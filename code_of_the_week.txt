from time import sleep
nums = [6,5,4,17,2,1,3,9,11,8]

print("Original List before sorting")
print(nums)
sleep(2)

print("We will now apply the Bubble Sort Algorithm")
sleep(1)


while True:
  sort = 0
  for x in range(len(nums)-1):
    
    if nums[x] > nums[x+1]:
      nums[x],nums[x+1] = nums[x+1],nums[x]
      sort = 1
    
    sleep(.1)
    print(nums)
  
  if sort == 0:
    break

print("Final List after bubble sort")
print(nums)
